#include <stdio.h>
#include <math.h>

struct rational{
	int num;
	int den;
};

struct rational add_rational(struct rational a,struct rational b){
    // function to add the rational numbers
	struct rational c;
	c.num = (a.num*b.den) + (b.num*a.den);
	c.den = (a.den*b.den);
	return c;
}

struct rational sub_rational(struct rational a,struct rational b){
    // function to subtract the rational numbers
	struct rational c;
	c.num = (a.num*b.den) - (b.num*a.den);
	c.den = (a.den*b.den);
	return c;
}

struct rational mul_rational(struct rational a,struct rational b){
    // function to multiply the rational numbers
	struct rational c;
	c.num = a.num*b.num;
	c.den = a.den*b.den;
	return c;
}

struct rational div_rational(struct rational a,struct rational b){
    // function to divide the rational numbers
	struct rational c;
	c.num = a.num*b.den;
	c.den = a.den*b.num;
	return c;
}

struct rational reduce_rational(struct rational a){
  // function to reduce the rational number to its lowest term
  int i;
  if (a.num > a.den){
	for (i=2;i<sqrt(a.den);i++){
       if (a.den%i == 0){
       	  if (a.num%i == 0){
       	  	while (a.den%i == 0 && a.num%i == 0){
       	  	a.den = a.den/i;
       	  	a.num = a.num/i;
       	     }
       	  }
       }
	}
	}
  else{
	for (i=2;i<sqrt(a.num);i++){
       if (a.num%i == 0){
       	  if (a.den%i == 0){
       	  	while(a.num%i == 0 && a.den%i == 0){
       	  	a.num = a.num/i;
       	  	a.den = a.den/i;
       	     }
       	  }
       }
	}
	}
	return a;
}
